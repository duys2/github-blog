## 리눅스 명령어들을 정리해 본다.

### shell

```shell
# 지정된 경로로 디렉토리 이동
cd [디렉토리 경로]

# 인자 없이 사용하면 사용자의 홈 디렉토리로 이동
cd

# 사용자의 홈 디렉토리로 이동 ('~'는 홈 디렉토리 의미)
cd ~

# 현재 디렉토리의 부모 디렉토리로 이동 ('..'는 부모 디렉토리 의미)
cd ..

# 이전에 있었던 디렉토리로 이동 ('-'는 이전 디렉토리 의미)
cd -

# 현재 작업 중인 디렉토리의 절대 경로 출력
pwd

# 현재 디렉토리의 파일과 디렉토리 목록 표시
ls

# 자세한 정보를 포함한 긴 형식으로 목록 표시 (파일 이름, 권한, 소유자, 크기, 수정 날짜 등)
ls -l

# 숨겨진 파일을 포함한 모든 파일과 디렉토리를 표시 ('.'으로 시작하는 파일 포함)
ls -a

# 파일 크기를 사람이 읽기 쉬운 형식으로 표시 (예: K, M, G)
ls -h

# 하위 디렉토리의 내용을 포함하여 모든 파일과 디렉토리를 재귀적으로 표시
ls -R

# 수정 시간을 기준으로 파일을 정렬 (가장 최신 파일 먼저 표시)
ls -t

# 역순으로 목록을 표시 (다른 옵션과 함께 사용 시 해당 정렬의 역순)
ls -r

# 숨겨진 파일을 포함한 모든 파일의 상세 정보를 사람이 읽기 쉬운 크기 형식으로 표시
ls -lha # 조합해서 사용 가능

# 파일, 디렉토리 이동 및 이름 변경
mv [옵션] 원본 대상

# 파일 또는 디렉토리 복사 (동일한 이름의 파일이 존재하면 덮어씀)
cp [옵션] 원본 대상

# 디렉토리 생성
mkdir [옵션] 디렉토리명

# '비어있는' 디렉토리 삭제
rmdir [옵션] [대상 디렉토리명]

# 파일, 디렉토리 삭제 (-f 옵션으로 강제 삭제)
rm [옵션] [대상]

# 파일의 내용 화면에 출력
cat 파일명

# 여러 파일의 내용을 연결하여 출력
cat 파일1 파일2 파일3

# 파일의 내용을 출력하면서 줄 번호 표시 (-n 옵션)
cat -n 파일명

# 파일의 내용을 출력하면서 탭과 줄 끝 문자를 표시 (-T, -E 옵션)
cat -TE 파일명

# 표준 입력에서 읽은 내용을 파일에 저장 (> 리디렉션 사용)
cat > 새파일명

# 파일 내용 정렬(오름차순)
sort 파일명

# 숫자를 기준으로 정렬 (-n 옵션)
sort -n 파일명

# 역순(내림차순)으로 정렬 (-r 옵션)
sort -r 파일명

# 중복된 라인 제거 (-u 옵션)
sort -u 파일명

# 지정 필드(열)를 기준으로 정렬 (-k 옵션)
sort -k2 파일명  # 두 번째 필드를 기준으로 정렬

# 대소문자를 구분하지 않고 정렬 (-f 옵션)
sort -f 파일명

# 파일에서 특정 패턴 검색
grep "검색어" 파일명

# 여러 파일에서 패턴 검색
grep "검색어" 파일1 파일2

# 현재 디렉토리의 모든 파일에서 패턴 검색
grep "검색어" *

# 대소문자 구분 없이 검색 (-i 옵션)
grep -i "검색어" 파일명

# 매치되는 라인 번호 표시 (-n 옵션)
grep -n "검색어" 파일명

# 패턴과 일치하지 않는 라인 출력 (-v 옵션)
grep -v "검색어" 파일명

# 재귀적으로 디렉토리 검색 (-r 또는 -R 옵션)
grep -r "검색어" 디렉토리명

# 지정된 경로에서 파일이나 디렉토리 검색
find [경로] [표현식]

# /home 디렉토리에서 "test.txt" 파일 찾기
find /home -name "test.txt"

# /home 디렉토리에서 "test.txt" 파일을 대소문자 구분 없이 찾기
find /home -iname "test.txt"

# /home 디렉토리에서 모든 디렉토리 찾기 (f는 일반 파일)
find /home -type d

# /home 디렉토리에서 최근 7일 이내에 수정된 파일과 디렉토리 찾기
find /home -mtime -7

# /home 디렉토리에서 "text.log" 파일을 찾아 "test.log"로 이름 변경
find /home -name text.log -exec mv {} test.log \;

# /home 디렉토리에서 .log 파일을 찾아 삭제
find /home -name "text.log" -exec rm {} \;

# 'newfile.txt'라는 새 파일 생성
touch newfile.txt

# 지정된 명령어의 실행 파일 위치 찾기
which [옵션] 명령어 # e.g. which java

# 현재 디렉토리의 모든 .txt 파일들을 나열
ls *.txt

# 한 문자로 된 이름을 가진 모든 .txt 파일 삭제하기
rm ?.txt

# a.txt, b.txt, c.txt 파일을 new_directory로 복사하기
cp [abc].txt new_directory/ # [abc].txt == a.txt, b.txt, c.txt

# Text.txt 파일과 동일한 내용을 가진 "hardlink"라는 새로운 파일 생성
ln Text.txt hardlink # 마치 하나의 파일을 두 개의 이름으로 부르는 것

# Text.txt 파일을 가리키는 "symlink"라는 심볼릭 링크 생성
ln -s Text.txt symlink # 화살표처럼 파일 위치를 가리키는 역할
```
